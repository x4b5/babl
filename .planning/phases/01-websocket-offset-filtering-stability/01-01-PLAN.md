<objective>
Implement WebSocket auto-reconnection with heartbeat monitoring for reliable real-time streaming transcription.

Purpose: Users experience silent WebSocket failures during 30-60 minute recording sessions. The connection dies without recovery, losing the live transcription stream. This plan adds automatic reconnection (up to 5 attempts with exponential backoff), server-side heartbeat to detect dead connections, and a user-visible reconnection status banner.

Output: Modified backend WebSocket endpoint with heartbeat, frontend with ReconnectingWebSocket and reconnect banner UI, npm dependency installed.

  ```typescript
  function startRealtimeStream() {
  	partialText = '';
  	liveSegments = [];
  	liveWorking = false;

  	const wsUrl = LOCAL_BACKEND_URL.replace('http', 'ws') + '/ws/transcribe-stream';
  	streamSocket = new ReconnectingWebSocket(wsUrl, [], {
  		maxRetries: 5,
  		maxReconnectionDelay: 10000,
  		minReconnectionDelay: 1000,
  		reconnectionDelayGrowFactor: 1.3,
  		connectionTimeout: 4000,
  		minUptime: 5000
  	});

  	streamSocket.addEventListener('open', () => {
  		console.log('[RT] WebSocket connected');
  		reconnecting = false;
  		reconnectStatus = '';
  		streamSocket!.send(JSON.stringify({ lang }));
  	});

  	streamSocket.addEventListener('message', (event: MessageEvent) => {
  		const data = JSON.parse(event.data);

  		if (data.type === 'ping') {
  			streamSocket!.send(JSON.stringify({ type: 'pong' }));
  			return;
  		}

  		if (data.type === 'partial') {
  			partialText = [...liveSegments, data.text].join(' ');
  		} else if (data.type === 'final') {
  			liveSegments = [...liveSegments, data.text];
  			partialText = liveSegments.join(' ');
  			liveWorking = true;
  		} else if (data.type === 'error') {
  			console.error('[RT] Error:', data.message);
  			error = `Real-time fout: ${data.message}`;
  		}
  	});

  	streamSocket.addEventListener('error', () => {
  		console.error('[RT] WebSocket error');
  		if (streamSocket && streamSocket.retryCount >= 5) {
  			reconnecting = false;
  			reconnectStatus = '';
  			error =
  				'Verbinding verloren. Je opname is bewaard \u2014 gebruik Bestand Upload om alsnog te transcriberen.';
  		} else if (streamSocket) {
  			reconnecting = true;
  			reconnectStatus = `Verbinding herstellen (poging ${(streamSocket.retryCount || 0) + 1}/5)...`;
  		}
  	});

  	streamSocket.addEventListener('close', () => {
  		console.log('[RT] WebSocket closed');
  	});
  }
  ```

  Key points per user decisions:
  - Per D-04: `liveSegments` is NOT cleared on reconnect -- only cleared at start. On reconnect (new `open` event), a new AssemblyAI session starts server-side, new segments append to existing liveSegments (per WS-04)
  - Per D-02: Recording continues during reconnect, audio chunks are buffered by MediaRecorder
  - Per D-05: Buffered audio during gap is NOT retroactively transcribed
  - Per D-03: After 5 failures, show exact error message text
  - Per D-08/D-09: All messages in Dutch

  async function sendChunkToStream(blob: Blob) {
  	if (!streamSocket || streamSocket.readyState !== WebSocket.OPEN) return;
  	try {
  		const pcm = await toPcmInt16(blob);
  		streamSocket.send(pcm);

  }
