    ```python
    import pytest
    from main import split_into_chunks

    class TestChunkOverlap:
        """FEED-03: Text chunks overlap 50-100 words for context preservation."""

        def test_single_chunk_no_overlap(self):
            """Short text should return single chunk, no overlap needed."""
            text = "Dit is een korte tekst."
            chunks = split_into_chunks(text, max_words=400)
            assert len(chunks) == 1

        def test_multi_chunk_has_overlap(self):
            """Long text split into chunks should have overlapping words."""
            # Generate text >800 words to force 3+ chunks
            words = " ".join([f"woord{i}" for i in range(900)])
            text = ". ".join([words[i:i+100] for i in range(0, len(words), 100)])
            chunks = split_into_chunks(text, max_words=400, overlap_words=75)
            assert len(chunks) >= 2
            # Check overlap exists between consecutive chunks
            for i in range(len(chunks) - 1):
                words_current = chunks[i].split()
                words_next = chunks[i + 1].split()
                # Last N words of current chunk should appear at start of next
                overlap = set(words_current[-75:]) & set(words_next[:75])
                assert len(overlap) > 0, f"No overlap between chunk {i} and {i+1}"

        def test_overlap_default_is_75_words(self):
            """Default overlap should be 75 words (middle of 50-100 range)."""
            # This test verifies the default parameter
            import inspect
            sig = inspect.signature(split_into_chunks)
            params = sig.parameters
            assert "overlap_words" in params
            assert params["overlap_words"].default == 75

        def test_overlap_zero_gives_no_overlap(self):
            """overlap_words=0 should give sequential chunks (backward compat)."""
            text = "Eerste zin. " * 200 + "Tweede zin. " * 200
            chunks = split_into_chunks(text, max_words=100, overlap_words=0)
            assert len(chunks) >= 2
            # Verify chunks don't share content
            full_text = " ".join(chunks)
            # Total words should be roughly same as original (no duplication)
            orig_words = len(text.split())
            full_words = len(full_text.split())
            assert full_words <= orig_words * 1.05  # Allow 5% tolerance

        def test_overlap_preserves_sentence_boundaries(self):
            """Overlap should start at sentence boundaries where possible."""
            sentences = [f"Zin nummer {i} met wat extra woorden erbij." for i in range(50)]
            text = " ".join(sentences)
            chunks = split_into_chunks(text, max_words=50, overlap_words=20)
            assert len(chunks) >= 2
    ```
