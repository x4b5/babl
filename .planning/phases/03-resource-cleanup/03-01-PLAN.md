**Part A: Create `src/lib/utils/cleanup.ts` -- extractable cleanup logic**

Create a new file with three exported functions. These are pure utility functions that accept resource references and clean them up. This makes cleanup logic testable without needing Svelte component context.

```typescript
/**
 * Resource cleanup utilities for audio/network resources.
 * Extracted from +page.svelte for testability and reuse.
 * Used by beforeunload, pagehide, and $effect cleanup paths.
 */

/**
 * Stop media resources: MediaRecorder, MediaStream tracks, AudioContext, animation frame.
 * Safe to call multiple times (idempotent).
 */
export function cleanupMediaResources(refs: {
	mediaRecorder: MediaRecorder | undefined;
	stream: MediaStream | undefined;
	audioContext: AudioContext | undefined;
	analyser: { current: AnalyserNode | undefined };
	animationFrameId: { current: number | undefined };
}): void {
	// 1. Stop MediaRecorder if recording
	if (refs.mediaRecorder && refs.mediaRecorder.state === 'recording') {
		try {
			refs.mediaRecorder.stop();
		} catch {
			/* already stopped */
		}
	}

	// 2. Stop all media tracks (turns off mic LED)
	if (refs.stream) {
		refs.stream.getTracks().forEach((track) => track.stop());
	}

	// 3. Cancel animation frame
	if (refs.animationFrameId.current) {
		cancelAnimationFrame(refs.animationFrameId.current);
		refs.animationFrameId.current = undefined;
	}

	// 4. Close AudioContext (check state to avoid double-close error)
	if (refs.audioContext && refs.audioContext.state !== 'closed') {
		refs.audioContext.close();
	}
	refs.analyser.current = undefined;
}

/**
 * Abort active network requests and close WebSocket.
 * Safe to call multiple times (idempotent).
 */
export function cleanupNetworkResources(refs: {
	transcribeController: AbortController | undefined;
	correctionController: AbortController | undefined;
	liveChunkController: AbortController | undefined;
	apiPollController: AbortController | undefined;
	streamSocket: { close: () => void } | undefined;
}): void {
	// 1. Abort fetch/SSE requests (fastest -- signals backend immediately)
	if (refs.transcribeController) {
		refs.transcribeController.abort();
	}
	if (refs.correctionController) {
		refs.correctionController.abort();
	}
	if (refs.liveChunkController) {
		refs.liveChunkController.abort();
	}
	if (refs.apiPollController) {
		refs.apiPollController.abort();
	}

	// 2. Close WebSocket (sends close frame, stops reconnection attempts)
	if (refs.streamSocket) {
		refs.streamSocket.close();
	}
}

/**
 * Clear all active timers.
 * Safe to call multiple times (idempotent).
 */
export function cleanupTimers(refs: {
	timerInterval: ReturnType<typeof setInterval> | undefined;
	processingTimerInterval: ReturnType<typeof setInterval> | undefined;
	liveInterval: ReturnType<typeof setInterval> | undefined;
	countdownInterval: ReturnType<typeof setInterval> | undefined;
	streamStallTimer: ReturnType<typeof setTimeout> | undefined;
}): void {
	if (refs.timerInterval) clearInterval(refs.timerInterval);
	if (refs.processingTimerInterval) clearInterval(refs.processingTimerInterval);
	if (refs.liveInterval) clearInterval(refs.liveInterval);
	if (refs.countdownInterval) clearInterval(refs.countdownInterval);
	if (refs.streamStallTimer) clearTimeout(refs.streamStallTimer);
}
```

$effect(() => {
	function handleBeforeUnload(e: BeforeUnloadEvent) {
		// Show confirmation dialog if recording/processing/correcting (D-01)
		// No dialog if idle or preparing (D-02)
		if (status === 'recording' || status === 'processing' || status === 'correcting') {
			e.preventDefault();
			e.returnValue = ''; // Legacy support -- custom messages not shown by browsers since 2017
		}
		// Cleanup runs regardless of dialog result
		cleanupAllResources();
	}

	function handlePageHide() {
		// Mobile fallback -- no confirmation dialog possible, just cleanup
		cleanupAllResources();
	}

	window.addEventListener('beforeunload', handleBeforeUnload);
	window.addEventListener('pagehide', handlePageHide);

	return () => {
		window.removeEventListener('beforeunload', handleBeforeUnload);
		window.removeEventListener('pagehide', handlePageHide);
		// Component destroy path (RC-02) -- same cleanup as page unload
		cleanupAllResources();
	};
});
